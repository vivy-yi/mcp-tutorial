## 错误处理规范

### JSON-RPC 错误格式

```json
{
  "jsonrpc": "2.0",
  "id": 1,
  "error": {
    "code": -32600,
    "message": "Invalid Request",
    "data": {
      "details": "The JSON sent is not a valid Request object"
    }
  }
}
```

### 错误码定义

| 错误码 | 名称 | 说明 |
|--------|------|------|
| -32700 | Parse Error | JSON 解析失败 |
| -32600 | Invalid Request | 无效的请求 |
| -32601 | Method Not Found | 方法不存在 |
| -32602 | Invalid Params | 参数无效 |
| -32603 | Internal Error | 内部错误 |
| -32000 | Server Error | 服务端错误 |

## 错误处理示例

### 服务端错误处理

```python
from fastmcp import FastMCP
from mcp.types import ErrorData

app = FastMCP("Error Handling Server")


@app.tool()
def divide(a: float, b: float) -> float:
    """除法运算"""
    if b == 0:
        raise ValueError("除数不能为零")

    if not isinstance(a, (int, float)) or not isinstance(b, (int, float)):
        raise TypeError("参数必须是数字")

    return a / b


# 自定义错误处理
@app.error_handler()
async def handle_error(error: Exception) -> ErrorData:
    """统一错误处理"""
    if isinstance(error, ValueError):
        return ErrorData(
            code=-32602,
            message="Invalid Parameters",
            data={"reason": str(error)}
        )
    elif isinstance(error, TypeError):
        return ErrorData(
            code=-32602,
            message="Invalid Parameters",
            data={"reason": str(error)}
        )
    else:
        return ErrorData(
            code=-32603,
            message="Internal Error",
            data={"reason": "未知错误"}
        )
```

### 客户端错误处理

```python
from mcp import ClientSession
from mcp.types import Error


async def safe_call_tool(session: ClientSession, name: str, arguments: dict):
    """安全的工具调用"""
    try:
        result = await session.call_tool(name, arguments)
        return result
    except Error as e:
        # 处理 MCP 协议错误
        print(f"MCP Error: {e.code} - {e.message}")
        if e.data:
            print(f"Details: {e.data}")
        raise
    except Exception as e:
        # 处理网络等异常
        print(f"Network Error: {e}")
        raise
```

## 超时处理

```python
import asyncio
from mcp import ClientSession


async def call_with_timeout(session: ClientSession, tool_name: str, args: dict, timeout: float = 30.0):
    """带超时的工具调用"""
    try:
        result = await asyncio.wait_for(
            session.call_tool(tool_name, args),
            timeout=timeout
        )
        return result
    except asyncio.TimeoutError:
        print(f"Tool call timed out after {timeout} seconds")
        raise
```

## 重试机制

```python
import asyncio
from functools import wraps


def retry_on_error(max_retries: int = 3, delay: float = 1.0):
    """重试装饰器"""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            last_exception = None
            for attempt in range(max_retries):
                try:
                    return await func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    if attempt < max_retries - 1:
                        await asyncio.sleep(delay * (attempt + 1))
            raise last_exception
        return wrapper
    return decorator


# 使用示例
@retry_on_error(max_retries=3, delay=2.0)
async def call_mcp_tool(session: ClientSession, name: str, args: dict):
    return await session.call_tool(name, args)
```

## 资源不存在处理

```python
@app.resource("file://{filename}")
def read_file(filename: str) -> str:
    """读取文件"""
    import os

    # 检查文件是否存在
    if not os.path.exists(filename):
        raise FileNotFoundError(f"文件不存在: {filename}")

    # 检查权限
    if not os.access(filename, os.R_OK):
        raise PermissionError(f"无权限读取文件: {filename}")

    with open(filename, 'r') as f:
        return f.read()
```
