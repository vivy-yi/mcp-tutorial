## 第一个 Go MCP 服务器

### 使用官方 SDK

```go
// cmd/server/main.go
package main

import (
	"context"
	"encoding/json"
	"fmt"
	"log"
	"os"

	"github.com/modelcontextprotocol/go-sdk/server"
	"github.com/modelcontextprotocol/go-sdk/transport/stdio"
)

func main() {
	// 创建服务器实例
	srv := server.NewServer(&server.Config{
		Name:    "hello-server",
		Version: "1.0.0",
	})

	// 注册工具
	srv.RegisterTool("greet", &server.Tool{
		Description: "向用户发送问候",
		InputSchema: map[string]interface{}{
			"type": "object",
			"properties": map[string]interface{}{
				"name": map[string]interface{}{
					"type":        "string",
					"description": "用户的名字",
				},
			},
			"required": []string{"name"},
		},
		Handler: func(ctx context.Context, args map[string]interface{}) (interface{}, error) {
			name, ok := args["name"].(string)
			if !ok {
				return nil, fmt.Errorf("name must be a string")
			}
			return fmt.Sprintf("你好，%s！欢迎使用 MCP！", name), nil
		},
	})

	// 创建传输层
	transport := stdio.NewStdioServerTransport(os.Stdin, os.Stdout)

	// 启动服务器
	log.Println("Starting MCP Server...")
	if err := srv.Run(context.Background(), transport); err != nil {
		log.Fatalf("Server error: %v", err)
	}
}
```

```go
// internal/tools/calculator.go
package tools

import (
	"context"
	"encoding/json"
	"errors"
	"math"
)

// Calculator 工具
type Calculator struct{}

func NewCalculator() *Calculator {
	return &Calculator{}
}

// ToolDefinition 返回工具定义
func (c *Calculator) ToolDefinition() map[string]interface{} {
	return map[string]interface{}{
		"name":        "calculate",
		"description": "执行数学运算",
		"inputSchema": map[string]interface{}{
			"type": "object",
			"properties": map[string]interface{}{
				"a": map[string]interface{}{
					"type":        "number",
					"description": "第一个操作数",
				},
				"b": map[string]interface{}{
					"type":        "number",
					"description": "第二个操作数",
				},
				"operation": map[string]interface{}{
					"type":        "string",
					"enum":        []string{"add", "subtract", "multiply", "divide", "power"},
					"description": "运算类型",
				},
			},
			"required": []string{"a", "b", "operation"},
		},
	}
}

// Handle 处理工具调用
func (c *Calculator) Handle(ctx context.Context, args map[string]interface{}) (interface{}, error) {
	a, ok := args["a"].(float64)
	if !ok {
		return nil, errors.New("参数 a 必须是数字")
	}

	b, ok := args["b"].(float64)
	if !ok {
		return nil, errors.New("参数 b 必须是数字")
	}

	operation, ok := args["operation"].(string)
	if !ok {
		return nil, errors.New("参数 operation 必须是字符串")
	}

	var result float64
	switch operation {
	case "add":
		result = a + b
	case "subtract":
		result = a - b
	case "multiply":
		result = a * b
	case "divide":
		if b == 0 {
			return nil, errors.New("除数不能为零")
		}
		result = a / b
	case "power":
		result = math.Pow(a, b)
	default:
		return nil, fmt.Errorf("不支持的运算: %s", operation)
	}

	return map[string]interface{}{
		"a":        a,
		"b":        b,
		"operation": operation,
		"result":   result,
	}, nil
}
```

### HTTP 请求工具

```go
// internal/tools/http.go
package tools

import (
	"bytes"
	"context"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"time"
)

// HTTPClient HTTP 客户端工具
type HTTPClient struct {
	client *http.Client
}

func NewHTTPClient() *HTTPClient {
	return &HTTPClient{
		client: &http.Client{
			Timeout: 30 * time.Second,
		},
	}
}

func (h *HTTPClient) ToolDefinition() map[string]interface{} {
	return map[string]interface{}{
		"name":        "http_request",
		"description": "发送 HTTP 请求",
		"inputSchema": map[string]interface{}{
			"type": "object",
			"properties": map[string]interface{}{
				"url": map[string]interface{}{
					"type":        "string",
					"description": "目标 URL",
				},
				"method": map[string]interface{}{
					"type":        "string",
					"enum":        []string{"GET", "POST", "PUT", "DELETE", "PATCH"},
					"description": "HTTP 方法",
					"default":     "GET",
				},
				"headers": map[string]interface{}{
					"type":        "object",
					"description": "请求头",
				},
				"body": map[string]interface{}{
					"type":        "object",
					"description": "请求体 (JSON)",
				},
			},
			"required": []string{"url"},
		},
	}
}

func (h *HTTPClient) Handle(ctx context.Context, args map[string]interface{}) (interface{}, error) {
	url, ok := args["url"].(string)
	if !ok {
		return nil, errors.New("url 必须是字符串")
	}

	method := "GET"
	if m, ok := args["method"].(string); ok {
		method = m
	}

	// 构建请求
	var body io.Reader
	if b, ok := args["body"].(map[string]interface{}); ok {
		jsonBody, err := json.Marshal(b)
		if err != nil {
			return nil, fmt.Errorf("序列化请求体失败: %v", err)
		}
		body = bytes.NewReader(jsonBody)
	}

	req, err := http.NewRequestWithContext(ctx, method, url, body)
	if err != nil {
		return nil, fmt.Errorf("创建请求失败: %v", err)
	}

	// 设置请求头
	if headers, ok := args["headers"].(map[string]interface{}); ok {
		for k, v := range headers {
			req.Header.Set(k, fmt.Sprintf("%v", v))
		}
	}

	// 发送请求
	resp, err := h.client.Do(req)
	if err != nil {
		return nil, fmt.Errorf("请求失败: %v", err)
	}
	defer resp.Body.Close()

	// 读取响应
	respBody, err := io.ReadAll(resp.Body)
	if err != nil {
		return nil, fmt.Errorf("读取响应失败: %v", err)
	}

	return map[string]interface{}{
		"status_code": resp.StatusCode,
		"headers":     resp.Header,
		"body":        string(respBody),
	}, nil
}
```

```go
// internal/resources/config.go
package resources

import (
	"context"
	"encoding/json"
	"os"
)

// ConfigResource 配置资源
type ConfigResource struct{}

func NewConfigResource() *ConfigResource {
	return &ConfigResource{}
}

// ResourceDefinition 返回资源定义
func (r *ConfigResource) ResourceDefinition() map[string]interface{} {
	return map[string]interface{}{
		"uri":         "config://app",
		"name":        "Application Config",
		"description": "应用程序配置",
		"mimeType":    "application/json",
	}
}

// Read 读取资源内容
func (r *ConfigResource) Read(ctx context.Context, uri string) (map[string]interface{}, error) {
	if uri != "config://app" {
		return nil, ErrResourceNotFound
	}

	config := map[string]interface{}{
		"version":     "1.0.0",
		"environment": getEnv("ENVIRONMENT", "production"),
		"features": map[string]bool{
			"auth":     true,
			"logging":  true,
			"debug":    getEnv("DEBUG", "false") == "true",
		},
	}

	return config, nil
}

func getEnv(key, defaultValue string) string {
	if value := os.Getenv(key); value != "" {
		return value
	}
	return defaultValue
}
```

### 错误处理中间件

```go
// internal/middleware/error.go
package middleware

import (
	"context"
	"log"

	"github.com/modelcontextprotocol/go-sdk/server"
)

// ErrorHandler 错误处理中间件
func ErrorHandler(next server.ToolHandler) server.ToolHandler {
	return func(ctx context.Context, args map[string]interface{}) (interface{}, error) {
		result, err := next(ctx, args)
		if err != nil {
			log.Printf("Tool error: %v", err)
			// 可以在这里添加更多错误处理逻辑
			// 例如：记录日志、发送告警等
			return map[string]interface{}{
				"error":   true,
				"message": err.Error(),
			}, nil // 返回错误信息而不是错误，让客户端能够处理
		}
		return result, nil
	}
}
```

## 并发处理

### Goroutine 并发工具

```go
// internal/tools/async.go
package tools

import (
	"context"
	"sync"
)

// AsyncBatch 异步批处理工具
type AsyncBatch struct {
	concurrency int
}

func NewAsyncBatch(concurrency int) *AsyncBatch {
	return &AsyncBatch{
		concurrency: concurrency,
	}
}

func (a *AsyncBatch) ToolDefinition() map[string]interface{} {
	return map[string]interface{}{
		"name":        "async_batch",
		"description": "异步批量处理任务",
		"inputSchema": map[string]interface{}{
			"type": "object",
			"properties": map[string]interface{}{
				"tasks": map[string]interface{}{
					"type":        "array",
					"description": "任务列表",
					"items": map[string]interface{}{
						"type": "object",
					},
				},
			},
			"required": []string{"tasks"},
		},
	}
}

func (a *AsyncBatch) Handle(ctx context.Context, args map[string]interface{}) (interface{}, error) {
	tasks, ok := args["tasks"].([]interface{})
	if !ok {
		return nil, ErrorInvalidParams
	}

	// 创建工作池
	taskChan := make(chan interface{}, len(tasks))
	resultChan := make(chan map[string]interface{}, len(tasks))

	// 启动工作协程
	var wg sync.WaitGroup
	for i := 0; i < a.concurrency; i++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			for task := range taskChan {
				// 处理单个任务
				result := processTask(task)
				resultChan <- result
			}
		}()
	}

	// 发送任务
	for _, task := range tasks {
		select {
		case taskChan <- task:
		case <-ctx.Done():
			return nil, ctx.Err()
		}
	}
	close(taskChan)

	// 等待完成
	go func() {
		wg.Wait()
		close(resultChan)
	}()

	// 收集结果
	results := make([]map[string]interface{}, 0, len(tasks))
	for result := range resultChan {
		results = append(results, result)
	}

	return map[string]interface{}{
		"total":  len(tasks),
		"result": results,
	}, nil
}

func processTask(task interface{}) map[string]interface{} {
	// 实际的任务处理逻辑
	return map[string]interface{}{
		"task":  task,
		"status": "completed",
	}
}
```

```go
// internal/tools/calculator_test.go
package tools

import (
	"context"
	"testing"

func TestCalculator(t *testing.T) {
	calc := NewCalculator()

	tests := []struct {
		name     string
		a        float64
		b        float64
		op       string
		expected float64
		err      bool
	}{
		{"加法", 5, 3, "add", 8, false},
		{"减法", 10, 4, "subtract", 6, false},
		{"乘法", 3, 4, "multiply", 12, false},
		{"除法", 10, 2, "divide", 5, false},
		{"除零", 10, 0, "divide", 0, true},
		{"幂运算", 2, 3, "power", 8, false},
	}

	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			args := map[string]interface{}{
				"a": tt.a,
				"b": tt.b,
				"op": tt.op,
			}

			result, err := calc.Handle(context.Background(), args)

			if tt.err {
				if err == nil {
					t.Error("期望错误，但未发生")
				}
				return
			}

			if err != nil {
				t.Errorf("发生意外错误: %v", err)
				return
			}

			resultMap := result.(map[string]interface{})
			if resultMap["result"] != tt.expected {
				t.Errorf("期望 %v，得到 %v", tt.expected, resultMap["result"])
			}
		})
	}
}
```

## 性能优化

### 连接池

```go
// internal/pool/connection.go
package pool

import (
	"sync"
)

// Pool 连接池
type Pool struct {
	mu       sync.Mutex
	items    []interface{}
	capacity int
	new      func() (interface{}, error)
}

func New(capacity int, newFunc func() (interface{}, error)) *Pool {
	return &Pool{
		items:    make([]interface{}, 0, capacity),
		capacity: capacity,
		new:      newFunc,
	}
}

func (p *Pool) Get() (interface{}, error) {
	p.mu.Lock()
	defer p.mu.Unlock()

	if len(p.items) > 0 {
		item := p.items[len(p.items)-1]
		p.items = p.items[:len(p.items)-1]
		return item, nil
	}

	return p.new()
}

func (p *Pool) Put(item interface{}) {
	p.mu.Lock()
	defer p.mu.Unlock()

	if len(p.items) < p.capacity {
		p.items = append(p.items, item)
	}
}
```
