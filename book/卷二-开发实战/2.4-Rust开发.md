## 第一个 Rust MCP 服务器

```rust
// src/main.rs
use model_context_protocol::{
    server::Server,
    types::{Tool, ToolCall, ToolCallResult, JsonSchema},
};
use serde::{Deserialize, Serialize};
use std::sync::Arc;
use tokio::sync::Mutex;

#[derive(Debug, Serialize, Deserialize)]
struct GreetArgs {
    name: String,
    greeting: Option<String>,
}

#[derive(Default)]
struct AppState {
    // 应用状态
}

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    // 初始化日志
    tracing_subscriber::fmt::init();

    // 创建服务器
    let state = Arc::new(Mutex::new(AppState::default()));
    let mut server = Server::new("hello-server", "1.0.0", state);

    // 注册工具
    server.register_tool(
        "greet",
        "向用户发送问候",
        JsonSchema::object()
            .property("name", JsonSchema::string().description("用户名字"))
            .property("greeting", JsonSchema::string().description("问候语"))
            .required(&["name"]),
        |state, args: GreetArgs| async move {
            let greeting = args.greeting.unwrap_or_else(|| "你好".to_string());
            let message = format!("{}，{}！欢迎使用 MCP！", greeting, args.name);

            Ok(ToolCallResult::Text(message))
        },
    )?;

    // 运行服务器（stdio 模式）
    server.run_stdio().await?;

    Ok(())
}
```

## 工具定义

### 基础工具

```rust
// src/tools/mod.rs
use model_context_protocol::{
    server::Server,
    types::{Tool, ToolCallResult, JsonSchema},
};
use serde::{Deserialize, Serialize};
use std::sync::Arc;
use tokio::sync::Mutex;

// 计算器工具参数
#[derive(Debug, Deserialize)]
struct CalculatorArgs {
    a: f64,
    b: f64,
    operation: String,
}

// 计算器工具结果
#[derive(Debug, Serialize)]
struct CalculatorResult {
    a: f64,
    b: f64,
    operation: String,
    result: Option<f64>,
    error: Option<String>,
}

pub fn register_calculator_tools<S: Default + 'static>(
    server: &mut Server<S>,
) -> Result<(), Box<dyn std::error::Error>> {
    server.register_tool(
        "calculate",
        "执行数学运算",
        JsonSchema::object()
            .property("a", JsonSchema::number().description("第一个操作数"))
            .property("b", JsonSchema::number().description("第二个操作数"))
            .property(
                "operation",
                JsonSchema::string()
                    .enum_values(&["add", "subtract", "multiply", "divide"])
                    .description("运算类型"),
            )
            .required(&["a", "b", "operation"]),
        |_state, args: CalculatorArgs| async move {
            let result = match args.operation.as_str() {
                "add" => Ok(args.a + args.b),
                "subtract" => Ok(args.a - args.b),
                "multiply" => Ok(args.a * args.b),
                "divide" => {
                    if args.b == 0.0 {
                        Err("除数不能为零".to_string())
                    } else {
                        Ok(args.a / args.b)
                    }
                }
                _ => Err(format!("不支持的运算: {}", args.operation)),
            };

            match result {
                Ok(value) => Ok(ToolCallResult::Json(serde_json::json!({
                    "a": args.a,
                    "b": args.b,
                    "operation": args.operation,
                    "result": value
                }))),
                Err(e) => Ok(ToolCallResult::Json(serde_json::json!({
                    "error": e
                }))),
            }
        },
    )?;

    Ok(())
}
```

### HTTP 客户端工具

```rust
// src/tools/http.rs
use model_context_protocol::{
    server::Server,
    types::{ToolCallResult, JsonSchema},
};
use reqwest::Client;
use serde::{Deserialize, Serialize};
use std::sync::Arc;
use tokio::sync::Mutex;

#[derive(Debug, Deserialize)]
struct HttpArgs {
    url: String,
    method: Option<String>,
    headers: Option<std::collections::HashMap<String, String>>,
    body: Option<serde_json::Value>,
}

#[derive(Debug, Serialize)]
struct HttpResult {
    status: u16,
    headers: std::collections::HashMap<String, String>,
    body: String,
}

pub fn register_http_tools<S: Default + 'static>(
    server: &mut Server<S>,
) -> Result<(), Box<dyn std::error::Error>> {
    let client = Arc::new(Mutex::new(Client::new()));

    server.register_tool(
        "http_request",
        "发送 HTTP 请求",
        JsonSchema::object()
            .property("url", JsonSchema::string().description("目标 URL"))
            .property(
                "method",
                JsonSchema::string()
                    .enum_values(&["GET", "POST", "PUT", "DELETE"])
                    .description("HTTP 方法"),
            )
            .property(
                "headers",
                JsonSchema::object()
                    .additional_properties(JsonSchema::string())
                    .description("请求头"),
            )
            .property("body", JsonSchema::any().description("请求体"))
            .required(&["url"]),
        move |_state, args: HttpArgs| async move {
            let client = client.lock().await;

            let mut request = match args.method.as_deref() {
                Some("POST") => client.post(&args.url),
                Some("PUT") => client.put(&args.url),
                Some("DELETE") => client.delete(&args.url),
                _ => client.get(&args.url),
            };

            // 添加请求头
            if let Some(headers) = args.headers {
                for (key, value) in headers {
                    request = request.header(&key, &value);
                }
            }

            // 添加请求体
            if let Some(body) = args.body {
                request = request.json(&body);
            }

            // 发送请求
            match request.send().await {
                Ok(response) => {
                    let status = response.status().as_u16();
                    let headers: std::collections::HashMap<String, String> = response
                        .headers()
                        .iter()
                        .map(|(k, v)| (k.to_string(), v.to_str().unwrap_or("").to_string()))
                        .collect();
                    let body = response.text().await.unwrap_or_default();

                    Ok(ToolCallResult::Json(serde_json::json!({
                        "status": status,
                        "headers": headers,
                        "body": body
                    })))
                }
                Err(e) => Ok(ToolCallResult::Json(serde_json::json!({
                    "error": e.to_string()
                }))),
            }
        },
    )?;

    Ok(())
}
```

## 错误处理

```rust
// src/error.rs
use serde::{Deserialize, Serialize};

#[derive(Debug, Serialize, Deserialize)]
pub struct MCPError {
    pub code: String,
    pub message: String,
    pub data: Option<serde_json::Value>,
}

impl MCPError {
    pub fn new(code: &str, message: &str) -> Self {
        Self {
            code: code.to_string(),
            message: message.to_string(),
            data: None,
        }
    }

    pub fn with_data(mut self, data: serde_json::Value) -> Self {
        self.data = Some(data);
        self
    }

    pub fn invalid_params(message: &str) -> Self {
        Self::new("INVALID_PARAMS", message)
    }

    pub fn not_found(message: &str) -> Self {
        Self::new("NOT_FOUND", message)
    }

    pub fn internal(message: &str) -> Self {
        Self::new("INTERNAL_ERROR", message)
    }
}

impl std::fmt::Display for MCPError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        write!(f, "{}: {}", self.code, self.message)
    }
}

impl std::error::Error for MCPError {}
```

// src/concurrent/mod.rs
use tokio::sync::mpsc;
use std::sync::Arc;
use tokio::sync::Mutex;

// 工作池
pub struct WorkerPool<T> {
    sender: mpsc::Sender<T>,
}

impl<T: Send + 'static> WorkerPool<T> {
    pub fn new(workers: usize) -> Self {
        let (sender, mut receiver) = mpsc::channel::<T>(100);

        // 创建工作协程
        for _ in 0..workers {
            let mut receiver = receiver.clone();
            tokio::spawn(async move {
                while let Some(job) = receiver.recv().await {
                    Self::process_job(job).await;
                }
            });
        }

        Self { sender }
    }

    async fn process_job(_job: T) {
        // 处理逻辑
    }

    pub async fn submit(&self, job: T) {
        let _ = self.sender.send(job).await;
    }
}

// 令牌桶限流
pub struct TokenBucket {
    tokens: Arc<Mutex<usize>>,
    capacity: usize,
    refill_rate: usize,
}

impl TokenBucket {
    pub fn new(capacity: usize, refill_rate: usize) -> Self {
        Self {
            tokens: Arc::new(Mutex::new(capacity)),
            capacity,
            refill_rate,
        }
    }

    pub async fn try_acquire(&self) -> bool {
        let mut tokens = self.tokens.lock().await;
        if *tokens > 0 {
            *tokens -= 1;
            true
        } else {
            false
        }
    }

    pub async fn refill(&self) {
        let mut tokens = self.tokens.lock().await;
        *tokens = (*tokens + self.refill_rate).min(self.capacity);
    }
}
