## MCP 集成测试架构

```python
# tests/integration/conftest.py
import pytest
import asyncio
from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client


@pytest.fixture
def server_params():
    """MCP 服务器参数"""
    return StdioServerParameters(
        command="python",
        args=["-m", "my_mcp_server"],
        env=None
    )


@pytest.fixture
async def session(server_params):
    """创建测试会话"""
    async with stdio_client(server_params) as (read, write):
        async with ClientSession(read, write) as session:
            await session.initialize()
            yield session
```

## 工具调用集成测试

```python
# tests/integration/test_tools.py
import pytest


class TestToolIntegration:
    """工具集成测试"""

    async def test_initialize(self, session):
        """测试初始化"""
        result = await session.initialize()
        assert result.protocolVersion is not None
        assert result.capabilities is not None

    async def test_list_tools(self, session):
        """测试列出工具"""
        result = await session.list_tools()
        tools = result.tools

        assert len(tools) > 0
        assert any(t.name == "add" for t in tools)

    async def test_call_tool(self, session):
        """测试调用工具"""
        result = await session.call_tool(
            name="add",
            arguments={"a": 5, "b": 3}
        )

        # 解析结果
        assert result.content[0].type == "text"
        assert "8" in result.content[0].text

    async def test_tool_error(self, session):
        """测试工具错误处理"""
        result = await session.call_tool(
            name="invalid_tool",
            arguments={}
        )

        assert result.isError


class TestToolChaining:
    """工具链测试"""

    async def test_multiple_tool_calls(self, session):
        """测试多次工具调用"""
        # 第一次调用
        result1 = await session.call_tool(
            name="add",
            arguments={"a": 5, "b": 3}
        )

        # 第二次调用（使用前一次结果）
        result2 = await session.call_tool(
            name="add",
            arguments={"a": int(result1.content[0].text), "b": 2}
        )

        assert "10" in result2.content[0].text
```

## 资源集成测试

```python
# tests/integration/test_resources.py
import pytest


class TestResourceIntegration:
    """资源集成测试"""

    async def test_list_resources(self, session):
        """测试列出资源"""
        result = await session.list_resources()
        resources = result.resources

        assert len(resources) > 0

    async def test_read_resource(self, session):
        """测试读取资源"""
        result = await session.read_resource(
            uri="config://app"
        )

        assert result.contents[0].mimeType == "application/json"

    async def test_read_invalid_resource(self, session):
        """测试读取不存在的资源"""
        with pytest.raises(Exception):
            await session.read_resource(uri="invalid://resource")
```

import pytest


class TestPromptIntegration:
    """提示模板集成测试"""

## 完整场景测试

```python
# tests/integration/test_scenarios.py
import pytest


class TestCompleteScenario:
    """完整场景测试"""

    async def test_github_workflow(self, session):
        """测试 GitHub 工作流"""
        # 1. 列出仓库
        repos = await session.call_tool(
            name="list_repositories",
            arguments={"org": "test-org"}
        )

        # 2. 获取第一个仓库的 PR
        prs = await session.call_tool(
            name="list_pull_requests",
            arguments={"repo": "test-repo"}
        )

        # 3. 创建新 PR
        new_pr = await session.call_tool(
            name="create_pull_request",
            arguments={
                "repo": "test-repo",
                "title": "Test PR",
                "body": "Test description"
            }
        )

        assert new_pr.content[0].type == "text"
```
